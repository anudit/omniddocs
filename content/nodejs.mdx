## Installation

<Tabs items={[ 'bun', 'pnpm','yarn', 'npm']} groupId="selected-pkg-manager" persist>

    ```sh
    bun i @omnid/sdk
    ```

  </Tab>
  <Tab value="pnpm">

    ```sh
    pnpm i --save @omnid/sdk
    ```

  </Tab>
  <Tab value="yarn">

    ```sh
    yarn add @omnid/sdk
    ```

  </Tab>
  <Tab value="npm">

    ```sh
    npm i --save @omnid/sdk
    ```

  </Tab>
</Tabs>
